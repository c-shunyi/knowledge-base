# 常用命令

# 代理配置

function p_on() {

    export http_proxy=http://127.0.0.1:7897

    export https_proxy=$http_proxy

    echo -e "终端代理已开启。"

}

  

function p_off(){

    unset http_proxy https_proxy

    echo -e "终端代理已关闭。"

}

# 快捷指令

### ====== 目录跳转 ======

alias ..='cd ..'              # 回到上一级目录

alias ...='cd ../..'          # 回到上两级目录

alias ....='cd ../../..'      # 回到上三级目录

alias -- -='cd -'              # 返回上一次所在的目录

alias home='cd ~'             # 快速回到用户主目录

alias desk='cd ~/Desktop'     # 快速进入桌面目录

alias dl='cd ~/Downloads'     # 快速进入下载目录

  

### ====== 文件/目录操作 ======

alias ll='ls -lh'             # 列出文件（显示详细信息+可读文件大小）

alias la='ls -lha'            # 列出所有文件（包含隐藏文件）

alias l='ls -CF'              # 简洁文件列表（带文件类型标记）

alias mkdir='mkdir -pv'       # 递归创建目录（如果不存在），并显示创建过程

alias rmf='rm -rf'            # 强制删除文件/目录（慎用）

alias cpv='rsync -ah --progress' # 复制文件/目录并显示进度条

  

### ====== Git 常用 ======

alias gs='git status'         # 查看 Git 仓库状态

alias ga='git add .'          # 添加所有更改到暂存区

alias gc='git commit -m'      # 提交更改（需后跟提交信息）

alias gcm='git commit -m'     # 同 gc，方便习惯使用

alias gp='git pull'           # 拉取远程最新代码

alias gpo='git push origin'   # 推送代码到 origin 仓库（需后跟分支名）

alias gco='git checkout'      # 切换分支

alias gcb='git checkout -b'   # 创建并切换到新分支

alias gb='git branch'         # 查看本地分支列表

alias gcl='git clone'         # 克隆远程仓库

  

### ====== 系统管理 ======

alias reload='source ~/.zshrc'  # 重新加载终端配置文件

alias update='brew update && brew upgrade && brew cleanup' # 更新 Homebrew 及其安装的软件

alias h='history'               # 查看历史命令

alias cls='clear'               # 清屏

  

### ====== 开发常用 ======

alias npms='npm start'         # 启动 npm 项目

alias npmi='npm install'       # 安装 npm 依赖

alias npmb='npm run build'     # 构建 npm 项目

alias p='pnpm'                 # 缩写 pnpm

alias y='yarn'                 # 缩写 yarn

  

### ====== NVM 快捷 ======

alias nvml='nvm list'          # 查看已安装的 Node 版本

alias nvmu='nvm use'           # 切换 Node 版本（需后跟版本号）

alias nvmi='nvm install'       # 安装指定 Node 版本
